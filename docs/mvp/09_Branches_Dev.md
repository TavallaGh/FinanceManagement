**[معیارهای پذیرش]{dir="rtl"} (Acceptance Criteria)**

-   [صفحه فقط برای کاربران دارای دسترسی]{dir="rtl"} [به فرم شعبه ها در
    تنظیمات سیستم، قابل مشاهده است؛ در غیر این صورت صفحه «دسترسی
    غیرمجاز» نمایش داده شود]{dir="rtl"}.

-   [ستون‌ها: کد، عنوان شعبه، آدرس، کد تفصیل، پیش‌فرض، فعال]{dir="rtl"}

-   [فیلترهای پیشرفته: کد، عنوان، پیش‌فرض (همه / بله / خیر)]{dir="rtl"}

-   [دکمه‌های اصلی: ایجاد جدید، حذف گروهی (چند انتخابی)]{dir="rtl"}

-   [در مودال ایجاد/ویرایش: فیلدهای کد\*، عنوان\*، آدرس، دو چک‌باکس
    «فعال» و «پیش‌فرض»]{dir="rtl"}

-   [هنگام انتخاب «پیش‌فرض» برای یک شعبه، سیستم به‌صورت خودکار تمام شعب
    دیگر را غیرپیش‌فرض می‌کند]{dir="rtl"}.

-   [ستون کد تفصیل]{dir="rtl"}:

    -   [اگر تخصیص داده شده باشد → نمایش بج آبی با امکان کلیک برای

    -   [اگر تخصیص نیافته باشد → نمایش «تخصیص نیافته» با دکمه

-   [مودال تخصیص کد تفصیل]{dir="rtl"}:

    -   [پیشنهاد خودکار شماره بعدی بر اساس]{dir="rtl"} [آخرین کد ایجاد
        شده برای [نوع تفصیل شعب]{.underline}]{dir="rtl"} [در صفحه
        تنظیمات شماره گذاری، تب شماره گذاری تفصیل ها]{dir="rtl"}

    -   [فیلد فقط به صورت عدد]{dir="rtl"}

-   [تغییر وضعیت فعال با چک‌باکس مستقیم در جدول (بدون نیاز به ذخیره

-   [دکمه‌های ویرایش و حذف در هر ردیف (طبق دسترسی)]{dir="rtl"}

-   [پس از هر عملیات موفق (ایجاد، ویرایش، حذف، تغییر وضعیت، تخصیص تفصیل)
