**[فرم مدیریت اشخاص و شرکت‌ها]{dir="rtl"}**

[این فرم **مرکز اطلاعات همه طرف‌های تجاری و پرسنل** شرکت است. هر شخص یا
شرکتی که در سیستم حسابداری استفاده می‌شود، باید اینجا ثبت
شود]{dir="rtl"}.

**[این فرم دقیقاً چه کاری انجام می‌دهد؟]{dir="rtl"}**

-   [ثبت اشخاص حقیقی (کارمند، مشتری، تأمین‌کننده و …)]{dir="rtl"}

-   [ثبت شرکت‌ها (حقوقی)]{dir="rtl"}

-   [تعریف چندین نقش همزمان برای هر شخص]{dir="rtl"}

-   [ثبت چندین آدرس (ممکن است یک طرف حساب آدرس های مختلفی داشته
    باشد)]{dir="rtl"}

-   [تخصیص کد تفصیل برای استفاده در اسناد حسابداری]{dir="rtl"}

**[نحوه کار با فرم]{dir="rtl"}**

**[[ایجاد شخص یا شرکت جدید]{dir="rtl"}]{.mark}**

1.  [دکمه «ایجاد جدید» را بزنید]{dir="rtl"}.

2.  [نوع را انتخاب کنید]{dir="rtl"}: [**حقیقی (شخص)** یا **حقوقی
    (شرکت)**]{dir="rtl"}.

3.  [فیلدهای اجباری را پر کنید]{dir="rtl"}:

    -   [حقیقی: نام، نام خانوادگی، کد ملی]{dir="rtl"}

    -   [حقوقی: نام شرکت، شناسه ملی/شماره ثبت]{dir="rtl"}

4.  [نقش‌ها را انتخاب کنید (می‌توانید چند نقش همزمان انتخاب
    کنید)]{dir="rtl"}.

5.  [آدرس(ها) را وارد کنید]{dir="rtl"}.

6.  [اطلاعات تکمیلی (تلفن، ایمیل، تاریخ تولد و …) را تکمیل
    کنید]{dir="rtl"}.

7.  «[ذخیره» را بزنید]{dir="rtl"}.

[**[تخصیص کد تفصیل]{.mark}**]{dir="rtl"}

[در ستون «کد تفصیل» روی عدد یا دکمه «تخصیص» کلیک کنید]{dir="rtl"}.
[سیستم شماره بعدی را پیشنهاد می‌دهد → ذخیره کنید]{dir="rtl"}.

**[[ویرایش و تغییر وضعیت]{dir="rtl"}]{.mark}**

-   [ویرایش: روی آیکون ویرایش کلیک کنید]{dir="rtl"}.

-   [فعال/غیرفعال کردن: تیک چک‌باکس ستون «فعال» را بزنید]{dir="rtl"}.

**[نکات خیلی مهم]{dir="rtl"}**

-   [هر شخص/شرکت فقط یک کد تفصیل می‌تواند داشته باشد]{dir="rtl"}.

-   [اگر شخصی غیرفعال شود، در فرم‌های سند حسابداری و انتخاب طرف حساب
    نمایش داده نمی‌شود]{dir="rtl"}.

-   [بهترین روش: ابتدا شخص را ثبت کنید، سپس نقش‌های مورد نیاز را به او
    بدهید]{dir="rtl"}.

-   [نقش‌ها در فیلترهای پیشرفته و گزارش‌ها بسیار مهم هستند (مثلاً گزارش
    تمام مشتریان یا تمام کارکنان)]{dir="rtl"}.
