**[راهنمای فرم مدیریت پروژه‌ها]{dir="rtl"}**

[این فرم برای ثبت و مدیریت تمام پروژه‌های شرکت طراحی شده است]{dir="rtl"}.

**[نحوه کار با فرم]{dir="rtl"}**

[**مشاهده لیست پروژه‌ها** بلافاصله پس از باز کردن فرم، لیست تمام پروژه‌ها
نمایش داده می‌شود]{dir="rtl"}. [با فیلترهای بالا (کد، عنوان، مدیر پروژه)
می‌توانید سریع‌تر پیدا کنید]{dir="rtl"}.

**[ایجاد پروژه جدید]{dir="rtl"}**

1.  [دکمه «ایجاد جدید» را بزنید]{dir="rtl"}.

2.  [کد و عنوان پروژه را وارد کنید (الزامی)]{dir="rtl"}.

3.  [تاریخ شروع و پایان را انتخاب کنید]{dir="rtl"}.

4.  [مدیر پروژه را از لیست انتخاب کنید]{dir="rtl"}.

5.  [بودجه پروژه را وارد کنید (اختیاری)]{dir="rtl"}.

6.  [وضعیت «فعال» را انتخاب کنید]{dir="rtl"}.

7.  «[ذخیره» را بزنید]{dir="rtl"}.

[**ویرایش پروژه**]{dir="rtl"}

[روی آیکون ویرایش در ردیف مورد نظر کلیک کنید → تغییرات را اعمال
کنید]{dir="rtl"}.

[**[تغییر وضعیت فعال/غیرفعال]{.mark}** کافی است تیک چک‌باکس ستون «فعال»
را بزنید یا بردارید]{dir="rtl"}.

[**[تخصیص کد تفصیل]{.mark}** در ستون «کد تفصیل» روی عدد یا دکمه «تخصیص»
کلیک کنید]{dir="rtl"}. [کد پیشنهادی نمایش داده می‌شود (یا کد قبلی را
ویرایش کنید) → «ذخیره]{dir="rtl"}».

[**[حذف پروژه]{.mark}** تیک ردیف(ها) را بزنید → دکمه حذف بالا را بزنید
یا روی آیکون سطل زباله در هر ردیف کلیک کنید]{dir="rtl"}.

**[نکات مهم]{dir="rtl"}**

-   [اگر دکمه‌ها یا گزینه‌ها را نمی‌بینید]{dir="rtl"} [به‌خاطر سطح دسترسی
    شماست]{dir="rtl"}.

-   [کد تفصیل معمولاً به‌صورت خودکار پیشنهاد می‌شود (مثلاً ۸۰۰۰۱، ۸۰۰۰۲ و
    …)]{dir="rtl"}.

-   [پس از هر تغییر، لیست به‌صورت خودکار بروز می‌شود]{dir="rtl"}.

-   [پروژه‌های غیرفعال در فرم‌های دیگر (مانند ثبت هزینه یا سند) نمایش داده
    نمی‌شوند]{dir="rtl"}.
