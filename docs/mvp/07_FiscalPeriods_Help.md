**[راهنمای فرم سال مالی و دوره‌های مالی]{dir="rtl"}**

[این فرم **قلب زمان‌بندی حسابداری** کل سیستم است]{dir="rtl"}. [تمام
اسناد، گزارش‌ها، بستن حساب‌ها و صورت‌های مالی بر اساس سال و دوره‌های
تعریف‌شده در اینجا کار می‌کنند]{dir="rtl"}.

**[تفاوت سال مالی و دوره مالی]{dir="rtl"}**

-   **[سال مالی]{dir="rtl"}**: [بازه زمانی کامل یک سال حسابداری (مثلاً
    ۱۴۰۴/۰۱/۰۱ تا ۱۴۰۴/۱۲/۲۹)]{dir="rtl"}.

-   **[دوره مالی]{dir="rtl"}**: [بازه‌های کوچکتر داخل سال (مثلاً ماهانه،
    فصلی یا شش‌ماهه)]{dir="rtl"}.

**[وضعیت‌های دوره مالی (خیلی مهم)]{dir="rtl"}**

-   **[باز نشده]{dir="rtl"} (Not Open)**: [هنوز هیچ سندی نمی‌توان ثبت
    کرد]{dir="rtl"}.

-   **[باز]{dir="rtl"} (Open)**: [می‌توان سند ثبت کرد]{dir="rtl"}.

-   **[بسته]{dir="rtl"} (Closed)**: [دیگر هیچ سندی نمی‌توان ثبت یا ویرایش
    کرد (حتی توسط مدیر سیستم)]{dir="rtl"}.

**[استثنائات دسترسی (ویژگی بسیار قدرتمند)]{dir="rtl"}**

[گاهی نیاز است یک کاربر خاص (مثلاً حسابدار ارشد) حتی در دوره بسته هم
بتواند سند ثبت کند]{dir="rtl"}. [در این فرم می‌توانید برای یک کاربر خاص
در یک دوره خاص، وضعیت‌های مجاز را تعریف کنید (مثلاً فقط بتواند سند «باز»
ببیند یا در دوره بسته هم سند ثبت کند)]{dir="rtl"}.

**[نحوه کار با فرم]{dir="rtl"}**

**[[تعریف سال مالی جدید]{dir="rtl"}]{.mark}**

1.  [دکمه ایجاد سال جدید را بزنید]{dir="rtl"}.

2.  [کد، عنوان، تاریخ شروع و پایان و نوع تقویم (شمسی یا میلادی) را وارد
    کنید]{dir="rtl"}.

[**[تولید خودکار دوره‌ها]{.mark}**]{dir="rtl"}

[پس از تعریف سال، روی سال کلیک کنید → دکمه‌های «ماهانه»، «فصلی» یا
«شش‌ماهه» را بزنید]{dir="rtl"}. [سیستم تمام دوره‌ها را به‌صورت خودکار ایجاد
می‌کند]{dir="rtl"}.

[**[تغییر وضعیت دوره]{.mark}**]{dir="rtl"}

[هر دوره را می‌توانید از «باز نشده» به «باز» و سپس به «بسته» تغییر
دهید]{dir="rtl"}.

**[[تعریف استثنا برای کاربر خاص]{dir="rtl"}]{.mark}**

1.  [روی یک دوره کلیک کنید]{dir="rtl"}.

2.  [پنل استثنائات سمت راست باز می‌شود]{dir="rtl"}.

3.  [کاربر را جستجو کنید]{dir="rtl"}.

4.  [وضعیت‌های مجاز برای او را تیک بزنید]{dir="rtl"}.

5.  [ذخیره کنید]{dir="rtl"}.

**[[نکات حیاتی]{dir="rtl"}]{.mark}**

-   [وقتی دوره‌ای را «بسته» کردید، هیچ کاربری (حتی مدیر سیستم) نمی‌تواند
    در آن دوره سند ثبت کند، مگر اینکه برای او استثنا تعریف کرده
    باشید]{dir="rtl"}.

-   [قبل از بستن سال مالی، حتماً مطمئن شوید همه اسناد ثبت و تأیید
    شده‌اند]{dir="rtl"}.

-   [تولید خودکار دوره‌ها فقط وقتی امکان‌پذیر است که هیچ دوره‌ای در وضعیت
    باز یا بسته نباشد]{dir="rtl"}.

[این فرم مستقیماً روی گردش کار، گزارش‌گیری سالانه و امنیت ثبت اسناد تأثیر
دارد]{dir="rtl"}. [اگر نیاز به مثال عملی بیشتری دارید (مثلاً چگونگی تعریف
استثنا برای حسابدار ارشد)، بگویید تا اضافه کنم]{dir="rtl"}.
