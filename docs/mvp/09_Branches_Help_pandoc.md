**[راهنمای فرم مدیریت شعب]{dir="rtl"}**

[این فرم برای ثبت و مدیریت تمام شعب شرکت طراحی شده است]{dir="rtl"}.

**[نحوه کار با فرم]{dir="rtl"}:**

**[مشاهده لیست شعب]{dir="rtl"}**

-   [بلافاصله پس از باز کردن فرم، لیست تمام شعب نمایش داده
    می‌شود]{dir="rtl"}.

-   [می‌توانید با فیلترهای بالا (کد، عنوان، پیش‌فرض) سریع‌تر پیدا
    کنید]{dir="rtl"}.

**[ایجاد شعبه جدید]{dir="rtl"}**

1.  [دکمه «ایجاد جدید» (بالای جدول) را بزنید]{dir="rtl"}.

2.  [کد شعبه و عنوان را وارد کنید (الزامی)]{dir="rtl"}.

3.  [آدرس را بنویسید (اختیاری)]{dir="rtl"}.

4.  [وضعیت «فعال» و «پیش‌فرض» را انتخاب کنید]{dir="rtl"}.

5.  [\"ذخیره\" را بزنید]{dir="rtl"}.

**[ویرایش شعبه]{dir="rtl"}**

-   [روی آیکون ویرایش در ردیف مورد نظر کلیک کنید]{dir="rtl"}.

-   [تغییرات را اعمال و ذخیره کنید]{dir="rtl"}.

**[تعیین شعبه پیش‌فرض]{dir="rtl"}**

-   [روی آیکون تیک سبز در ستون «پیش‌فرض» کلیک کنید]{dir="rtl"}.

-   [سیستم به‌طور خودکار شعبه قبلی را غیرپیش‌فرض می‌کند]{dir="rtl"}.

**[تغییر وضعیت فعال/غیرفعال]{dir="rtl"}**

-   [کافی است تیک چک‌باکس ستون «فعال» را بزنید یا بردارید]{dir="rtl"}.

**[تخصیص کد تفصیل]{dir="rtl"}**

-   [در ستون «کد تفصیل» روی عدد یا دکمه «تخصیص» کلیک کنید]{dir="rtl"}.

-   [کد پیشنهادی نمایش داده می‌شود (یا کد قبلی را ویرایش
    کنید)]{dir="rtl"}.

-   [دکمه ذخیره را بزنید]{dir="rtl"}.

**[حذف شعبه]{dir="rtl"}**

-   [تیک ردیف(ها) را بزنید → دکمه حذف بالا را بزنید]{dir="rtl"}.

-   [یا روی آیکون سطل زباله در هر ردیف کلیک کنید]{dir="rtl"}.

**[نکات مهم]{dir="rtl"}**

-   [اگر دکمه‌ها یا گزینه‌ها را نمی‌بینید → به‌خاطر سطح دسترسی شماست (با
    مدیر سیستم هماهنگ کنید)]{dir="rtl"}.

-   [فقط یک شعبه می‌تواند «پیش‌فرض» باشد]{dir="rtl"}.

-   [کد تفصیل معمولاً به‌صورت خودکار پیشنهاد می‌شود (مثلاً ۰۰۱، ۰۰۲ و
    ...)]{dir="rtl"}.

-   [پس از هر تغییر، لیست به‌صورت خودکار بروز می‌شود]{dir="rtl"}.
