**[راهنمای فرم مدیریت مراکز هزینه]{dir="rtl"}**

[این فرم برای ثبت و مدیریت تمام مراکز هزینه شرکت طراحی شده
است]{dir="rtl"}.

**[نحوه کار با فرم]{dir="rtl"}:**

**[مشاهده لیست مراکز هزینه]{dir="rtl"}**

-   [بلافاصله پس از باز کردن فرم، لیست تمام مراکز هزینه نمایش داده
    می‌شود]{dir="rtl"}.

-   [با فیلترهای بالا (کد، عنوان، نوع مرکز) می‌توانید سریع‌تر پیدا
    کنید]{dir="rtl"}.

**[ایجاد مرکز هزینه جدید]{dir="rtl"}**

1.  [دکمه «ایجاد جدید» (بالای جدول) را بزنید]{dir="rtl"}.

2.  [کد و عنوان را وارد کنید (الزامی)]{dir="rtl"}.

3.  [نوع مرکز را انتخاب کنید (تولیدی / خدماتی / اداری)]{dir="rtl"}.

4.  [آدرس یا توضیحات را بنویسید (اختیاری)]{dir="rtl"}.

5.  [وضعیت «فعال» را انتخاب کنید]{dir="rtl"}.

6.  [«ذخیره» را بزنید]{dir="rtl"}.

**[ویرایش مرکز هزینه (فقط در صورتی که هیچ سندی برای مرکز مورد نظر ثبت
نشده باشد)]{dir="rtl"}**

-   [روی آیکون ویرایش در ردیف مورد نظر کلیک کنید]{dir="rtl"}.

-   [تغییرات را اعمال و ذخیره کنید]{dir="rtl"}.

**[تغییر وضعیت فعال/غیرفعال]{dir="rtl"}**

-   [کافی است تیک چک‌باکس ستون «فعال» را بزنید یا بردارید]{dir="rtl"}.

**[تخصیص کد تفصیل]{dir="rtl"}**

-   [در ستون «کد تفصیل» روی عدد یا دکمه «تخصیص» کلیک کنید]{dir="rtl"}.

-   [کد پیشنهادی نمایش داده می‌شود (یا کد قبلی را ویرایش
    کنید)]{dir="rtl"}.

-   [«ذخیره» را بزنید]{dir="rtl"}.

**[حذف مرکز هزینه (فقط در صورتی که هیچ سندی برای مرکز مورد نظر ثبت نشده
باشد)]{dir="rtl"}**

-   [تیک ردیف(ها) را بزنید → دکمه حذف بالا را بزنید]{dir="rtl"}.

-   [یا روی آیکون سطل زباله در هر ردیف کلیک کنید]{dir="rtl"}.

**[نکات مهم]{dir="rtl"}**

-   [اگر دکمه‌ها یا گزینه‌ها را نمی‌بینید → به‌خاطر سطح دسترسی شماست (با
    مدیر سیستم هماهنگ کنید)]{dir="rtl"}.

-   [کد تفصیل معمولاً به‌صورت خودکار پیشنهاد می‌شود (مثلاً ۹۰۰۱، ۹۰۰۲ و
    ...)]{dir="rtl"}.

-   [پس از هر تغییر، لیست به‌صورت خودکار بروز می‌شود]{dir="rtl"}.

-   [نوع مرکز فقط یکی از سه گزینه تولیدی، خدماتی یا اداری می‌تواند
    باشد]{dir="rtl"}.
